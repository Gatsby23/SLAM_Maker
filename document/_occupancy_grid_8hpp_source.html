37 template<typename PointT>
38 typename pcl::PointCloud<PointT>::Ptr OccupancyGrid::segmentCloud(
39  const typename pcl::PointCloud<PointT>::Ptr & cloudIn,
40  const pcl::IndicesPtr & indicesIn,
41  const Transform & pose,
42  const cv::Point3f & viewPoint,
43  pcl::IndicesPtr & groundIndices,
44  pcl::IndicesPtr & obstaclesIndices,
45  pcl::IndicesPtr * flatObstacles) const
46 {
47  typename pcl::PointCloud<PointT>::Ptr cloud(new pcl::PointCloud<PointT>);
48 
49  // voxelize to grid cell size
50  cloud = util3d::voxelize(cloudIn, indicesIn, cellSize_);
51  pcl::IndicesPtr indices(new std::vector<int>);
52  indices->resize(cloud->size());
53  for(unsigned int i=0; i<indices->size(); ++i)
54  {
55  indices->at(i) = i;
56  }
57 
58  // add pose rotation without yaw
59  float roll, pitch, yaw;
60  pose.getEulerAngles(roll, pitch, yaw);
61  UDEBUG("node.getPose()=%s projMapFrame_=%d", pose.prettyPrint().c_str(), projMapFrame_?1:0);
62  cloud = util3d::transformPointCloud(cloud, Transform(0,0, projMapFrame_?pose.z():0, roll, pitch, 0));
63 
64  // filter footprint
65  if(footprintLength_ > 0.0f || footprintWidth_ > 0.0f || footprintHeight_ > 0.0f)
66  {
67  indices = util3d::cropBox(
68  cloud,
69  indices,
70  Eigen::Vector4f(
71  footprintLength_>0.0f?-footprintLength_/2.0f:std::numeric_limits<int>::min(),
72  footprintWidth_>0.0f&&footprintLength_>0.0f?-footprintWidth_/2.0f:std::numeric_limits<int>::min(),
73  0,
74  1),
75  Eigen::Vector4f(
76  footprintLength_>0.0f?footprintLength_/2.0f:std::numeric_limits<int>::max(),
77  footprintWidth_>0.0f&&footprintLength_>0.0f?footprintWidth_/2.0f:std::numeric_limits<int>::max(),
78  footprintHeight_>0.0f&&footprintLength_>0.0f&&footprintWidth_>0.0f?footprintHeight_:std::numeric_limits<int>::max(),
79  1),
80  Transform::getIdentity(),
81  true);
82  }
83 
84  // filter ground/obstacles zone
85  if(minGroundHeight_ != 0.0f || maxObstacleHeight_ != 0.0f)
86  {
87  indices = util3d::passThrough(cloud, indices, "z",
88  minGroundHeight_!=0.0f?minGroundHeight_:std::numeric_limits<int>::min(),
89  maxObstacleHeight_>0.0f?maxObstacleHeight_:std::numeric_limits<int>::max());
90  UDEBUG("indices after max obstacles height filtering = %d", (int)indices->size());
91  }
92 
93  if(indices->size())
94  {
95  if(normalsSegmentation_ && !groundIsObstacle_)
96  {
97  UDEBUG("normalKSearch=%d", normalKSearch_);
98  UDEBUG("maxGroundAngle=%f", maxGroundAngle_);
99  UDEBUG("Cluster radius=%f", clusterRadius_);
100  UDEBUG("flatObstaclesDetected=%d", flatObstaclesDetected_?1:0);
101  UDEBUG("maxGroundHeight=%f", maxGroundHeight_);
102  util3d::segmentObstaclesFromGround<PointT>(
103  cloud,
104  indices,
105  groundIndices,
106  obstaclesIndices,
107  normalKSearch_,
108  maxGroundAngle_,
109  clusterRadius_,
110  minClusterSize_,
111  flatObstaclesDetected_,
112  maxGroundHeight_,
113  flatObstacles,
114  Eigen::Vector4f(viewPoint.x, viewPoint.y, viewPoint.z+(projMapFrame_?pose.z():0), 1));
115  UDEBUG("viewPoint=%f,%f,%f", viewPoint.x, viewPoint.y, viewPoint.z+(projMapFrame_?pose.z():0));
116  //UWARN("Saving ground.pcd and obstacles.pcd");
117  //pcl::io::savePCDFile("ground.pcd", *cloud, *groundIndices);
118  //pcl::io::savePCDFile("obstacles.pcd", *cloud, *obstaclesIndices);
119  }
120  else
121  {
122  UDEBUG("");
123  // passthrough filter
124  groundIndices = rtabmap::util3d::passThrough(cloud, indices, "z",
125  minGroundHeight_!=0.0f?minGroundHeight_:std::numeric_limits<int>::min(),
126  maxGroundHeight_!=0.0f?maxGroundHeight_:std::numeric_limits<int>::max());
127 
128  pcl::IndicesPtr notObstacles = groundIndices;
129  if(indices->size())
130  {
131  notObstacles = util3d::extractIndices(cloud, indices, true);
132  notObstacles = util3d::concatenate(notObstacles, groundIndices);
133  }
134  obstaclesIndices = rtabmap::util3d::extractIndices(cloud, notObstacles, true);
135  }
136 
137  UDEBUG("groundIndices=%d obstaclesIndices=%d", (int)groundIndices->size(), (int)obstaclesIndices->size());
138 
139  // Do radius filtering after voxel filtering ( a lot faster)
140  if(noiseFilteringRadius_ > 0.0 && noiseFilteringMinNeighbors_ > 0)
141  {
142  UDEBUG("");
143  if(groundIndices->size())
144  {
145  groundIndices = rtabmap::util3d::radiusFiltering(cloud, groundIndices, noiseFilteringRadius_, noiseFilteringMinNeighbors_);
146  }
147  if(obstaclesIndices->size())
148  {
149  obstaclesIndices = rtabmap::util3d::radiusFiltering(cloud, obstaclesIndices, noiseFilteringRadius_, noiseFilteringMinNeighbors_);
150  }
151  if(flatObstacles && (*flatObstacles)->size())
152  {
153  *flatObstacles = rtabmap::util3d::radiusFiltering(cloud, *flatObstacles, noiseFilteringRadius_, noiseFilteringMinNeighbors_);
154  }
155 
156  if(groundIndices->empty() && obstaclesIndices->empty())
157  {
158  UWARN("Cloud (with %d points) is empty after noise "
159  "filtering. Occupancy grid cannot be "
160  "created.",
161  (int)cloud->size());
162 
163  }
164  }
165  }
166  return cloud;
167 }
