136 template<class T>
137 Quaternion<T>::Quaternion(const T phi, const T theta, const T psi){
138  T sphi = sin(phi);
139  T stheta = sin(theta);
140  T spsi = sin(psi);
141  T cphi = cos(phi);
142  T ctheta = cos(theta);
143  T cpsi = cos(psi);
144 
145  T _r[3][3] = { //create rotational Matrix
146  {cpsi*ctheta, cpsi*stheta*sphi - spsi*cphi, cpsi*stheta*cphi + spsi*sphi},
147  {spsi*ctheta, spsi*stheta*sphi + cpsi*cphi, spsi*stheta*cphi - cpsi*sphi},
148  { -stheta, ctheta*sphi, ctheta*cphi}
149  };
150 
151  T _w = sqrt(MY_MAX(0, 1 + _r[0][0] + _r[1][1] + _r[2][2]))/2.0;
152  T _x = sqrt(MY_MAX(0, 1 + _r[0][0] - _r[1][1] - _r[2][2]))/2.0;
153  T _y = sqrt(MY_MAX(0, 1 - _r[0][0] + _r[1][1] - _r[2][2]))/2.0;
154  T _z = sqrt(MY_MAX(0, 1 - _r[0][0] - _r[1][1] + _r[2][2]))/2.0;
155  this->w = _w;
156  this->x = (_r[2][1] - _r[1][2])>=0?fabs(_x):-fabs(_x);
157  this->y = (_r[0][2] - _r[2][0])>=0?fabs(_y):-fabs(_y);
158  this->z = (_r[1][0] - _r[0][1])>=0?fabs(_z):-fabs(_z);
159 }

214 template<class T>
215 inline Vector3<T> Quaternion<T>::toAngles() const{
216  T n = this->norm();
217  T s = n > 0?2./(n*n):0.;
218 
219  T m00, m10, m20, m21, m22;
220  T phi,theta,psi;
221 
222 
223  T xs = this->x*s;
224  T ys = this->y*s;
225  T zs = this->z*s;
226 
227  T wx = this->w*xs;
228  T wy = this->w*ys;
229  T wz = this->w*zs;
230 
231  T xx = this->x*xs;
232  T xy = this->x*ys;
233  T xz = this->x*zs;
234 
235  T yy = this->y*ys;
236  T yz = this->y*zs;
237 
238  T zz = this->z*zs;
239 
240  m00 = 1.0 - (yy + zz);
241  //m11 = 1.0 - (xx + zz);
242  m22 = 1.0 - (xx + yy);
243 
244 
245  m10 = xy + wz;
246  //m01 = xy - wz;
247 
248  m20 = xz - wy;
249  //m02 = xz + wy;
250  m21 = yz + wx;
251  //m12 = yz - wx;
252 
253  phi = atan2(m21,m22);
254  theta = atan2(-m20,sqrt(m21*m21 + m22*m22));
255  psi = atan2(m10,m00);
256 
257  return Vector3<T>(phi, theta, psi);
258 }

333 template<class T>
334 inline Quaternion<T> slerp(const Quaternion<T>& from, const Quaternion<T>& to, const T lambda){
335  Quaternion<T> _from = from.normalized();
336  Quaternion<T> _to = to.normalized();
337  T _cos_omega = innerproduct(_from,_to);
338  _cos_omega = (_cos_omega>1)?1:_cos_omega;
339  _cos_omega = (_cos_omega<-1)?-1:_cos_omega;
340  T _omega = acos(_cos_omega);
341  assert (!isnan(_cos_omega));
342  if (fabs(_omega) < 1e-6)
343  return to;
344 
345  //determine right direction of slerp:
346  Quaternion<T> _pq = _from - _to;
347  Quaternion<T> _pmq = _from + _to;
348  T _first = _pq.norm();
349  T _alternativ = _pmq.norm();
350 
351  Quaternion<T> q1 = _from;
352  Quaternion<T> q2 = (_first < _alternativ)? (Quaternion<T>) _to: -1.*(Quaternion<T>)_to;
353  //now calculate intermediate quaternion.
354 
355  Quaternion<T> ret = q1*(sin((1-lambda)*_omega)/(sin(_omega))) + q2*(sin(lambda*_omega)/sin(_omega));
356  assert (!(isnan(ret.w) || isnan(ret.x) || isnan(ret.y) || isnan(ret.z)));
357  return ret;
358 }
