61 template <class E>
62 struct EVComparator{
64  enum CompareMode {CompareLevel, CompareLength};
65  CompareMode mode;
66 
67  EVComparator(){
68  mode=CompareLevel;
69  }
70  inline bool operator() (const E& e1, const E& e2){
71  int o1=0, o2=0;
72  switch (mode){
73  case CompareLevel:
74  o1=e1->top->level;
75  o2=e2->top->level;
76  break;
77  case CompareLength:
78  o1=e1->length;
79  o2=e2->length;
80  break;
81  }
82  return o1<o2;
83  }
